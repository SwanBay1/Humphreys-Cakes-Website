  <script>
    // Footer year
    document.getElementById("year").textContent = new Date().getFullYear();

    /**
     * TEMPLATE CONFIG (easy to add/remove images)
     * Add/remove items in this array for each cake page.
     * Keep paths relative to your CarrotCake.html file.
     */
    const images = [
      "Assets/CarrotCake/CarrotCake.jpg",
      "Assets/CarrotCake/CarrotCake.jpg"
     
    ];

    // ---- Gallery logic ----
    const mainImage = document.getElementById("mainImage");
    const prevBtn = document.getElementById("prevBtn");
    const nextBtn = document.getElementById("nextBtn");
    const counterText = document.getElementById("counterText");
    const thumbRow = document.getElementById("thumbRow");

    let index = 0;

    function renderThumbs() {
      thumbRow.innerHTML = "";
      images.forEach((src, i) => {
        const t = document.createElement("button");
        t.type = "button";
        t.className = "thumb" + (i === index ? " is-active" : "");
        t.setAttribute("aria-label", `View image ${i + 1}`);
        t.addEventListener("click", () => {
          index = i;
          updateGallery();
        });

        const img = document.createElement("img");
        img.src = src;
        img.alt = "Thumbnail " + (i + 1);
        t.appendChild(img);

        thumbRow.appendChild(t);
      });
    }

    function updateButtons() {
      // If you prefer looping instead of disabling ends, change this logic.
      prevBtn.disabled = (index === 0);
      nextBtn.disabled = (index === images.length - 1);
    }

    function updateCounter() {
      counterText.textContent = `Image ${Math.min(index + 1, images.length)} of ${images.length}`;
    }

    function updateMainImage() {
      const safeIndex = Math.max(0, Math.min(index, images.length - 1));
      index = safeIndex;

      mainImage.src = images[index] || "";
      mainImage.alt = `Carrot Cake cake image ${index + 1}`;
    }

    function updateGallery() {
      updateMainImage();
      updateButtons();
      updateCounter();
      renderThumbs();
    }

    prevBtn.addEventListener("click", () => {
      if (index > 0) {
        index -= 1;
        updateGallery();
      }
    });

    nextBtn.addEventListener("click", () => {
      if (index < images.length - 1) {
        index += 1;
        updateGallery();
      }
    });

    // Keyboard support (left/right arrows)
    document.addEventListener("keydown", (e) => {
      if (e.key === "ArrowLeft") prevBtn.click();
      if (e.key === "ArrowRight") nextBtn.click();
    });

    // Initialise
    updateGallery();
  </script>
